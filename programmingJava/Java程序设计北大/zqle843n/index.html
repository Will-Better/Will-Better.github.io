输入与输出

约 403 字大约 1 分钟

2025-04-08

应用程序(Java Application)的输入输出可以是文本界面，也可以是图形界面。

1 文本界面: in和out

代码示例
// 导入java.io包
import java.io.*;
public class AppCharInOut  {
	public static void main(String[] args)  {
		char c = ' ';
		System.out.print("Please input a char: ");
        // 使用try{}catch避免出现错误
		try {
            //System.in.read( ) 读取
			c = (char) System.in.read();
		}catch(IOException e){}
        // System.out.print( ) 及 println、printf 打印
		System.out.println("You have entered: " + c );
	}
}
// 获取输入的行
BufferedReader in = new BufferedReader(
	new InputStreamReader( System.in ) );
// 读取一行
s = in.readLine();

// 获取输入的行
BufferedReader in = new BufferedReader(
	new InputStreamReader( System.in ) );
System.out.print("Please input an int: ");
s = in.readLine();
// 将码字符串转换为整数
n = Integer.parseInt( s );
System.out.print("Please input a double: ");
s = in.readLine();
// 将码字符串转换为双精度
d = Double.parseDouble( s );

2 图形界面

  • Java Application需要首先创建自己的图形界面.
  • 根据用户的输入及操作进行处理
代码示例
import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class AppGraphInOut {
	public static void main( String args[] ) {
		new AppFrame();
	}
}

class  AppFrame extends JFrame {
	JTextField in = new JTextField(10);
	JButton btn = new JButton("求平方");
	JLabel out = new JLabel("结果");
    // 构造函数
	public AppFrame() {
		setLayout(new FlowLayout());
		getContentPane().add(in);
		getContentPane().add(btn);
		getContentPane().add(out);
        // 在Java8前添加事件处理的方法
		btn.addActionListener(new BtnActionAdapter());
        // 在Java8后可简写为此类型
        btn.addActionListener( e->{
            // 获取文本框输入的字符串
			String s = in.getText();
			double d = Double.parseDouble( s );
			double sq = d * d;
            // 在label上展示文字
			out.setText( d + "平方后结果：" + sq );
		});
		setSize(400,100);
		setDefaultCloseOperation(DISPOSE_ON_CLOSE);
		setVisible(true);
	}

	class BtnActionAdapter implements ActionListener {
        // 具体处理事件
		public void actionPerformed( ActionEvent e ){
			String s = in.getText();
			double d = Double.parseDouble( s );
			double sq = d * d;
			out.setText( d + "平方后结果：" + sq );
		}
	}
}
